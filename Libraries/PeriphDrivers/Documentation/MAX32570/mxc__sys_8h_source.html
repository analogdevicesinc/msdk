244typedef struct {
245 int ie_status;
246 int in_critical;
247} mxc_crit_state_t;
248
249static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
250
251static inline void _mxc_crit_get_state()
252{
253#ifndef __riscv
254 /*
255 On ARM M the 0th bit of the Priority Mask register indicates
256 whether interrupts are enabled or not.
257
258 0 = enabled
259 1 = disabled
260 */
261 uint32_t primask = __get_PRIMASK();
262 _state.ie_status = (primask == 0);
263#else
264 /*
265 On RISC-V bit position 3 (Machine Interrupt Enable) of the
266 mstatus register indicates whether interrupts are enabled.
267
268 0 = disabled
269 1 = enabled
270 */
271 uint32_t mstatus = get_mstatus();
272 _state.ie_status = ((mstatus & (1 << 3)) != 0);
273#endif
274}
275
287static inline void MXC_SYS_Crit_Enter(void)
288{
289 _mxc_crit_get_state();
290 if (_state.ie_status)
291 __disable_irq();
292 _state.in_critical = 1;
293}
294
299static inline void MXC_SYS_Crit_Exit(void)
300{
301 if (_state.ie_status) {
302 __enable_irq();
303 }
304 _state.in_critical = 0;
305 _mxc_crit_get_state();
306 /*
307 ^ Reset the state again to prevent edge case
308 where interrupts get disabled, then Crit_Exit() gets
309 called, which would inadvertently re-enable interrupts
310 from old state.
311 */
312}
313
319static inline int MXC_SYS_In_Crit_Section(void)
320{
321 return _state.in_critical;
322}
