120typedef struct {
121 int ie_status;
122 int in_critical;
123} mxc_crit_state_t;
124
125static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
126
127static inline void _mxc_crit_get_state()
128{
129#ifndef __riscv
130 /*
131 On ARM M the 0th bit of the Priority Mask register indicates
132 whether interrupts are enabled or not.
133
134 0 = enabled
135 1 = disabled
136 */
137 uint32_t primask = __get_PRIMASK();
138 _state.ie_status = (primask == 0);
139#else
140 /*
141 On RISC-V bit position 3 (Machine Interrupt Enable) of the
142 mstatus register indicates whether interrupts are enabled.
143
144 0 = disabled
145 1 = enabled
146 */
147 uint32_t mstatus = get_mstatus();
148 _state.ie_status = ((mstatus & (1 << 3)) != 0);
149#endif
150}
151
163static inline void MXC_SYS_Crit_Enter(void)
164{
165 _mxc_crit_get_state();
166 if (_state.ie_status)
167 __disable_irq();
168 _state.in_critical = 1;
169}
170
175static inline void MXC_SYS_Crit_Exit(void)
176{
177 if (_state.ie_status) {
178 __enable_irq();
179 }
180 _state.in_critical = 0;
181 _mxc_crit_get_state();
182 /*
183 ^ Reset the state again to prevent edge case
184 where interrupts get disabled, then Crit_Exit() gets
185 called, which would inadvertently re-enable interrupts
186 from old state.
187 */
188}
189
195static inline int MXC_SYS_In_Crit_Section(void)
196{
197 return _state.in_critical;
198}
