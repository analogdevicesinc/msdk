113typedef struct {
114 int ie_status;
115 int in_critical;
116} mxc_crit_state_t;
117
118static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
119
120static inline void _mxc_crit_get_state()
121{
122#ifndef __riscv
123 /*
124 On ARM M the 0th bit of the Priority Mask register indicates
125 whether interrupts are enabled or not.
126
127 0 = enabled
128 1 = disabled
129 */
130 uint32_t primask = __get_PRIMASK();
131 _state.ie_status = (primask == 0);
132#else
133 /*
134 On RISC-V bit position 3 (Machine Interrupt Enable) of the
135 mstatus register indicates whether interrupts are enabled.
136
137 0 = disabled
138 1 = enabled
139 */
140 uint32_t mstatus = get_mstatus();
141 _state.ie_status = ((mstatus & (1 << 3)) != 0);
142#endif
143}
144
148static inline void MXC_SYS_Crit_Enter(void)
149{
150 _mxc_crit_get_state();
151 if (_state.ie_status)
152 __disable_irq();
153 _state.in_critical = 1;
154}
155
160static inline void MXC_SYS_Crit_Exit(void)
161{
162 if (_state.ie_status) {
163 __enable_irq();
164 }
165 _state.in_critical = 0;
166 _mxc_crit_get_state();
167 /*
168 ^ Reset the state again to prevent edge case
169 where interrupts get disabled, then Crit_Exit() gets
170 called, which would inadvertently re-enable interrupts
171 from old state.
172 */
173}
174
180static inline int MXC_SYS_In_Crit_Section(void)
181{
182 return _state.in_critical;
183}
