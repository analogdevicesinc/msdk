157typedef struct {
158 int ie_status;
159 int in_critical;
160} mxc_crit_state_t;
161
162static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
163
164static inline void _mxc_crit_get_state()
165{
166#ifndef __riscv
167 /*
168 On ARM M the 0th bit of the Priority Mask register indicates
169 whether interrupts are enabled or not.
170
171 0 = enabled
172 1 = disabled
173 */
174 uint32_t primask = __get_PRIMASK();
175 _state.ie_status = (primask == 0);
176#else
177 /*
178 On RISC-V bit position 3 (Machine Interrupt Enable) of the
179 mstatus register indicates whether interrupts are enabled.
180
181 0 = disabled
182 1 = enabled
183 */
184 uint32_t mstatus = get_mstatus();
185 _state.ie_status = ((mstatus & (1 << 3)) != 0);
186#endif
187}
188
192static inline void MXC_SYS_Crit_Enter(void)
193{
194 _mxc_crit_get_state();
195 if (_state.ie_status)
196 __disable_irq();
197 _state.in_critical = 1;
198}
199
204static inline void MXC_SYS_Crit_Exit(void)
205{
206 if (_state.ie_status) {
207 __enable_irq();
208 }
209 _state.in_critical = 0;
210 _mxc_crit_get_state();
211 /*
212 ^ Reset the state again to prevent edge case
213 where interrupts get disabled, then Crit_Exit() gets
214 called, which would inadvertently re-enable interrupts
215 from old state.
216 */
217}
218
224static inline int MXC_SYS_In_Crit_Section(void)
225{
226 return _state.in_critical;
227}
