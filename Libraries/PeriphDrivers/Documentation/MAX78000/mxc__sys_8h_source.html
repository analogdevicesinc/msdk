205typedef struct {
206 int ie_status;
207 int in_critical;
208} mxc_crit_state_t;
209
210static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
211
212static inline void _mxc_crit_get_state()
213{
214#ifndef __riscv
215 /*
216 On ARM M the 0th bit of the Priority Mask register indicates
217 whether interrupts are enabled or not.
218
219 0 = enabled
220 1 = disabled
221 */
222 uint32_t primask = __get_PRIMASK();
223 _state.ie_status = (primask == 0);
224#else
225 /*
226 On RISC-V bit position 3 (Machine Interrupt Enable) of the
227 mstatus register indicates whether interrupts are enabled.
228
229 0 = disabled
230 1 = enabled
231 */
232 uint32_t mstatus = get_mstatus();
233 _state.ie_status = ((mstatus & (1 << 3)) != 0);
234#endif
235}
236
248static inline void MXC_SYS_Crit_Enter(void)
249{
250 _mxc_crit_get_state();
251 if (_state.ie_status)
252 __disable_irq();
253 _state.in_critical = 1;
254}
255
260static inline void MXC_SYS_Crit_Exit(void)
261{
262 if (_state.ie_status) {
263 __enable_irq();
264 }
265 _state.in_critical = 0;
266 _mxc_crit_get_state();
267 /*
268 ^ Reset the state again to prevent edge case
269 where interrupts get disabled, then Crit_Exit() gets
270 called, which would inadvertently re-enable interrupts
271 from old state.
272 */
273}
274
280static inline int MXC_SYS_In_Crit_Section(void)
281{
282 return _state.in_critical;
283}
