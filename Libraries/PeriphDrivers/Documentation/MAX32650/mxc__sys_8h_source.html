245typedef struct {
246 int ie_status;
247 int in_critical;
248} mxc_crit_state_t;
249
250static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
251
252static inline void _mxc_crit_get_state()
253{
254#ifndef __riscv
255 /*
256 On ARM M the 0th bit of the Priority Mask register indicates
257 whether interrupts are enabled or not.
258
259 0 = enabled
260 1 = disabled
261 */
262 uint32_t primask = __get_PRIMASK();
263 _state.ie_status = (primask == 0);
264#else
265 /*
266 On RISC-V bit position 3 (Machine Interrupt Enable) of the
267 mstatus register indicates whether interrupts are enabled.
268
269 0 = disabled
270 1 = enabled
271 */
272 uint32_t mstatus = get_mstatus();
273 _state.ie_status = ((mstatus & (1 << 3)) != 0);
274#endif
275}
276
280static inline void MXC_SYS_Crit_Enter(void)
281{
282 _mxc_crit_get_state();
283 if (_state.ie_status)
284 __disable_irq();
285 _state.in_critical = 1;
286}
287
292static inline void MXC_SYS_Crit_Exit(void)
293{
294 if (_state.ie_status) {
295 __enable_irq();
296 }
297 _state.in_critical = 0;
298 _mxc_crit_get_state();
299 /*
300 ^ Reset the state again to prevent edge case
301 where interrupts get disabled, then Crit_Exit() gets
302 called, which would inadvertently re-enable interrupts
303 from old state.
304 */
305}
306
312static inline int MXC_SYS_In_Crit_Section(void)
313{
314 return _state.in_critical;
315}
