131typedef struct {
132 int ie_status;
133 int in_critical;
134} mxc_crit_state_t;
135
136static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
137
138static inline void _mxc_crit_get_state()
139{
140#ifndef __riscv
141 /*
142 On ARM M the 0th bit of the Priority Mask register indicates
143 whether interrupts are enabled or not.
144
145 0 = enabled
146 1 = disabled
147 */
148 uint32_t primask = __get_PRIMASK();
149 _state.ie_status = (primask == 0);
150#else
151 /*
152 On RISC-V bit position 3 (Machine Interrupt Enable) of the
153 mstatus register indicates whether interrupts are enabled.
154
155 0 = disabled
156 1 = enabled
157 */
158 uint32_t mstatus = get_mstatus();
159 _state.ie_status = ((mstatus & (1 << 3)) != 0);
160#endif
161}
162
174static inline void MXC_SYS_Crit_Enter(void)
175{
176 _mxc_crit_get_state();
177 if (_state.ie_status)
178 __disable_irq();
179 _state.in_critical = 1;
180}
181
186static inline void MXC_SYS_Crit_Exit(void)
187{
188 if (_state.ie_status) {
189 __enable_irq();
190 }
191 _state.in_critical = 0;
192 _mxc_crit_get_state();
193 /*
194 ^ Reset the state again to prevent edge case
195 where interrupts get disabled, then Crit_Exit() gets
196 called, which would inadvertently re-enable interrupts
197 from old state.
198 */
199}
200
206static inline int MXC_SYS_In_Crit_Section(void)
207{
208 return _state.in_critical;
209}
