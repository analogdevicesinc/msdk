185typedef struct {
186 int ie_status;
187 int in_critical;
188} mxc_crit_state_t;
189
190static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
191
192static inline void _mxc_crit_get_state()
193{
194#ifndef __riscv
195 /*
196 On ARM M the 0th bit of the Priority Mask register indicates
197 whether interrupts are enabled or not.
198
199 0 = enabled
200 1 = disabled
201 */
202 uint32_t primask = __get_PRIMASK();
203 _state.ie_status = (primask == 0);
204#else
205 /*
206 On RISC-V bit position 3 (Machine Interrupt Enable) of the
207 mstatus register indicates whether interrupts are enabled.
208
209 0 = disabled
210 1 = enabled
211 */
212 uint32_t mstatus = get_mstatus();
213 _state.ie_status = ((mstatus & (1 << 3)) != 0);
214#endif
215}
216
220static inline void MXC_SYS_Crit_Enter(void)
221{
222 _mxc_crit_get_state();
223 if (_state.ie_status)
224 __disable_irq();
225 _state.in_critical = 1;
226}
227
232static inline void MXC_SYS_Crit_Exit(void)
233{
234 if (_state.ie_status) {
235 __enable_irq();
236 }
237 _state.in_critical = 0;
238 _mxc_crit_get_state();
239 /*
240 ^ Reset the state again to prevent edge case
241 where interrupts get disabled, then Crit_Exit() gets
242 called, which would inadvertently re-enable interrupts
243 from old state.
244 */
245}
246
252static inline int MXC_SYS_In_Crit_Section(void)
253{
254 return _state.in_critical;
255}
