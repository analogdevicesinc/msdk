209typedef struct {
210 int ie_status;
211 int in_critical;
212} mxc_crit_state_t;
213
214static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
215
216static inline void _mxc_crit_get_state()
217{
218#ifndef __riscv
219 /*
220 On ARM M the 0th bit of the Priority Mask register indicates
221 whether interrupts are enabled or not.
222
223 0 = enabled
224 1 = disabled
225 */
226 uint32_t primask = __get_PRIMASK();
227 _state.ie_status = (primask == 0);
228#else
229 /*
230 On RISC-V bit position 3 (Machine Interrupt Enable) of the
231 mstatus register indicates whether interrupts are enabled.
232
233 0 = disabled
234 1 = enabled
235 */
236 uint32_t mstatus = get_mstatus();
237 _state.ie_status = ((mstatus & (1 << 3)) != 0);
238#endif
239}
240
252static inline void MXC_SYS_Crit_Enter(void)
253{
254 _mxc_crit_get_state();
255 if (_state.ie_status)
256 __disable_irq();
257 _state.in_critical = 1;
258}
259
264static inline void MXC_SYS_Crit_Exit(void)
265{
266 if (_state.ie_status) {
267 __enable_irq();
268 }
269 _state.in_critical = 0;
270 _mxc_crit_get_state();
271 /*
272 ^ Reset the state again to prevent edge case
273 where interrupts get disabled, then Crit_Exit() gets
274 called, which would inadvertently re-enable interrupts
275 from old state.
276 */
277}
278
284static inline int MXC_SYS_In_Crit_Section(void)
285{
286 return _state.in_critical;
287}
