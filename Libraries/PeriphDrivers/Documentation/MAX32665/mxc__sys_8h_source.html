220typedef struct {
221 int ie_status;
222 int in_critical;
223} mxc_crit_state_t;
224
225static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
226
227static inline void _mxc_crit_get_state()
228{
229#ifndef __riscv
230 /*
231 On ARM M the 0th bit of the Priority Mask register indicates
232 whether interrupts are enabled or not.
233
234 0 = enabled
235 1 = disabled
236 */
237 uint32_t primask = __get_PRIMASK();
238 _state.ie_status = (primask == 0);
239#else
240 /*
241 On RISC-V bit position 3 (Machine Interrupt Enable) of the
242 mstatus register indicates whether interrupts are enabled.
243
244 0 = disabled
245 1 = enabled
246 */
247 uint32_t mstatus = get_mstatus();
248 _state.ie_status = ((mstatus & (1 << 3)) != 0);
249#endif
250}
251
263static inline void MXC_SYS_Crit_Enter(void)
264{
265 _mxc_crit_get_state();
266 if (_state.ie_status)
267 __disable_irq();
268 _state.in_critical = 1;
269}
270
275static inline void MXC_SYS_Crit_Exit(void)
276{
277 if (_state.ie_status) {
278 __enable_irq();
279 }
280 _state.in_critical = 0;
281 _mxc_crit_get_state();
282 /*
283 ^ Reset the state again to prevent edge case
284 where interrupts get disabled, then Crit_Exit() gets
285 called, which would inadvertently re-enable interrupts
286 from old state.
287 */
288}
289
295static inline int MXC_SYS_In_Crit_Section(void)
296{
297 return _state.in_critical;
298}
