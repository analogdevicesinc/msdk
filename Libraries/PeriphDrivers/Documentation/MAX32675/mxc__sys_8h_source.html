170typedef struct {
171 int ie_status;
172 int in_critical;
173} mxc_crit_state_t;
174
175static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
176
177static inline void _mxc_crit_get_state()
178{
179#ifndef __riscv
180 /*
181 On ARM M the 0th bit of the Priority Mask register indicates
182 whether interrupts are enabled or not.
183
184 0 = enabled
185 1 = disabled
186 */
187 uint32_t primask = __get_PRIMASK();
188 _state.ie_status = (primask == 0);
189#else
190 /*
191 On RISC-V bit position 3 (Machine Interrupt Enable) of the
192 mstatus register indicates whether interrupts are enabled.
193
194 0 = disabled
195 1 = enabled
196 */
197 uint32_t mstatus = get_mstatus();
198 _state.ie_status = ((mstatus & (1 << 3)) != 0);
199#endif
200}
201
213static inline void MXC_SYS_Crit_Enter(void)
214{
215 _mxc_crit_get_state();
216 if (_state.ie_status)
217 __disable_irq();
218 _state.in_critical = 1;
219}
220
225static inline void MXC_SYS_Crit_Exit(void)
226{
227 if (_state.ie_status) {
228 __enable_irq();
229 }
230 _state.in_critical = 0;
231 _mxc_crit_get_state();
232 /*
233 ^ Reset the state again to prevent edge case
234 where interrupts get disabled, then Crit_Exit() gets
235 called, which would inadvertently re-enable interrupts
236 from old state.
237 */
238}
239
245static inline int MXC_SYS_In_Crit_Section(void)
246{
247 return _state.in_critical;
248}
